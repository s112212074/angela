    <script>
        function handleNavigation() {
            // 取得目前的 hash，如果沒有就預設為 '#home'
            const hash = window.location.hash || '#home';
            const targetId = hash.substring(1);
            
            // 隱藏所有章節
            const sections = document.querySelectorAll('.page-section');
            sections.forEach(sec => {
                sec.style.display = 'none';
            });
            
            // 顯示目標章節
            const targetSec = document.getElementById(targetId);
            if (targetSec) {
                // 如果是首頁，用 flex 以維持文字垂直置中的佈局；其他區塊用 block
                targetSec.style.display = targetId === 'home' ? 'flex' : 'block';
            }

            // 更新選單的活躍狀態 (Active State)
            const links = document.querySelectorAll('.nav-link');
            links.forEach(link => {
                if (link.getAttribute('href') === hash) {
                    link.classList.add('bg-brand-dark', 'text-brand-accent', 'border-brand-accent');
                    link.classList.remove('border-transparent');
                } else {
                    link.classList.remove('bg-brand-dark', 'text-brand-accent', 'border-brand-accent');
                    link.classList.add('border-transparent');
                }
            });
        }

        // 監聽網址 hash 變化 (點擊選單或上一頁/下一頁)
        window.addEventListener('hashchange', handleNavigation);
        // 頁面載入時執行一次
        window.addEventListener('DOMContentLoaded', handleNavigation);
    </script>
